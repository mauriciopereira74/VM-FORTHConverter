    expressions     : expressions Word
                    | word

    Word            : number
                    | function
                    | operator
                    | contition
                    | function_definition

    function_definition : ":" STRING operators ";"


    operator    : aritmetic
                | dot
                | oneOne
                | swap
                | drop
                | dup
                | over
                | rot
                | print
                | number
                | condition


    operators   : operators aritmetic
                | operators dot
                | operators oneOne
                | operators swap
                | operators drop
                | operators dup
                | operators over
                | operators rot
                | operators print
                | operators number
                | operators condition


    operators   : aritmetic
                | dot
                | oneOne
                | swap
                | drop
                | dup
                | over
                | rot
                | print
                | number
                | condition


    aritmetic   : OPERATORS
                | MOD

    print   : DOT
            | DO
            | EMIT
            | CHAR
            | SPACES
            | CR

    dup     : DUP

    condition   : if

    do      : DO do_operators LOOP

    else    : ELSE operators THEN

    OneOne  : ONEONE

    drop    : DROP

    rot     : ROT

    swap    : SWAP

    over    : OVER
